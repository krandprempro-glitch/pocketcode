```java
public class MainTabActivity extends AppCompatActivity {
    private TabLayout tabLayout;
    private ViewPager2 viewPager;
    private TabPagerAdapter pagerAdapter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main_tabs);
        
        initViews();
        setupTabs();
    }
    
    private void initViews() {
        tabLayout = findViewById(R.id.tab_layout);
        viewPager = findViewById(R.id.view_pager);
        
        pagerAdapter = new TabPagerAdapter(this);
        viewPager.setAdapter(pagerAdapter);
    }
    
    private void setupTabs() {
        new TabLayoutMediator(tabLayout, viewPager, (tab, position) -> {
            switch (position) {
                case 0:
                    tab.setText("SSH连接");
                    tab.setIcon(R.drawable.ic_ssh);
                    break;
                case 1:
                    tab.setText("文件浏览");
                    tab.setIcon(R.drawable.ic_folder_open);
                    break;
                case 2:
                    tab.setText("Git变更");
                    tab.setIcon(R.drawable.ic_git);
                    break;
                case 3:
                    tab.setText("设置");
                    tab.setIcon(R.drawable.ic_settings);
                    break;
            }
        }).attach();
        
        // 默认选中Tab2 (文件浏览)
        viewPager.setCurrentItem(1, false);
    }
}
```

### 3. 完善错误处理机制

#### 网络连接错误处理
```java
// SFTPConnectionManager.java 中的连接错误处理
public Observable<Boolean> connect(SSHConnectionConfig config) {
    return Observable.fromCallable(() -> {
        try {
            sshClient = new SSHClient();
            sshClient.addHostKeyVerifier(new PromiscuousVerifier());
            
            // 设置连接超时
            sshClient.setConnectTimeout(30000);
            sshClient.setTimeout(30000);
            
            sshClient.connect(config.getHost(), config.getPort());
            sshClient.authPassword(config.getUsername(), config.getPassword());
            
            sftpClient = sshClient.newSFTPClient();
            return true;
            
        } catch (ConnectException e) {
            throw new RuntimeException("网络连接失败: " + e.getMessage());
        } catch (UnknownHostException e) {
            throw new RuntimeException("主机地址无效: " + e.getMessage());
        } catch (UserAuthException e) {
            throw new RuntimeException("用户认证失败: " + e.getMessage());
        } catch (IOException e) {
            throw new RuntimeException("连接过程出错: " + e.getMessage());
        }
    }).subscribeOn(Schedulers.io())
      .observeOn(AndroidSchedulers.mainThread())
      .doOnError(this::handleConnectionError);
}

private void handleConnectionError(Throwable error) {
    Log.e(TAG, "SFTP连接错误", error);
    // 显示用户友好的错误消息
    showErrorDialog("连接失败", error.getMessage());
}
```

```java
public Observable<List<RemoteFileItem>> listFiles(String remotePath) {
    return Observable.fromCallable(() -> {
        try {
            List<RemoteFileItem> files = new ArrayList<>();
            for (RemoteResourceInfo resource : sftpClient.ls(remotePath)) {
                // 文件处理逻辑
            }
            return files;
            
        } catch (SFTPException e) {
            if (e.getStatusCode() == Response.StatusCode.NO_SUCH_FILE) {
                throw new RuntimeException("目录不存在: " + remotePath);
            } else if (e.getStatusCode() == Response.StatusCode.PERMISSION_DENIED) {
                throw new RuntimeException("权限不足，无法访问目录: " + remotePath);
            } else {
                throw new RuntimeException("无法读取目录内容: " + e.getMessage());
            }
        } catch (IOException e) {
            throw new RuntimeException("网络错误: " + e.getMessage());
        }
    }).subscribeOn(Schedulers.io())
      .observeOn(AndroidSchedulers.mainThread());
}
```

```java
private void showErrorDialog(String title, String message) {
    new AlertDialog.Builder(this)
            .setTitle(title)
            .setMessage(message)
            .setIcon(R.drawable.ic_error)
            .setPositiveButton("确定", null)

            })

}

private void handleError(Throwable error) {
    runOnUiThread(() -> {
        hideLoadingIndicator();
        
        String errorMessage = error.getMessage();
        if (errorMessage.contains("网络连接失败")) {
            showErrorDialog("网络错误", "请检查网络连接和服务器地址");
        } else if (errorMessage.contains("用户认证失败")) {
            showErrorDialog("认证错误", "用户名或密码错误，请检查登录凭据");
        } else if (errorMessage.contains("权限不足")) {
            showErrorDialog("权限错误", "当前用户没有足够权限访问此资源");
        } else {
            showErrorDialog("操作失败", errorMessage);
        }
    });
}
```

### 4. 连接状态管理

#### 连接状态监控
```java
// 连接状态枚举
public enum ConnectionStatus {
    DISCONNECTED,
    CONNECTING,
    CONNECTED,
    ERROR
}

// 连接状态管理
private BehaviorSubject<ConnectionStatus> connectionStatusSubject = 
    BehaviorSubject.createDefault(ConnectionStatus.DISCONNECTED);

public Observable<ConnectionStatus> getConnectionStatus() {
    return connectionStatusSubject.asObservable();
}

private void updateConnectionStatus(ConnectionStatus status) {
    connectionStatusSubject.onNext(status);
    updateUI(status);
}

private void updateUI(ConnectionStatus status) {
    runOnUiThread(() -> {
        switch (status) {
            case CONNECTING:
                connectionStatusIcon.setImageResource(R.drawable.ic_circle_yellow);
                connectionStatusText.setText("连接中...");
                break;
            case CONNECTED:
                connectionStatusIcon.setImageResource(R.drawable.ic_circle_green);
                connectionStatusText.setText("已连接");
                break;
            case DISCONNECTED:
            case ERROR:
                connectionStatusIcon.setImageResource(R.drawable.ic_circle_red);
                connectionStatusText.setText("已断开");
                break;
        }
    });
}
```

#### 文件下载权限检查
```java
private void checkDownloadPermission() {
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
        if (checkSelfPermission(Manifest.permission.WRITE_EXTERNAL_STORAGE) 
            != PackageManager.PERMISSION_GRANTED) {
            
            requestPermissions(
                new String[]{Manifest.permission.WRITE_EXTERNAL_STORAGE},
                REQUEST_STORAGE_PERMISSION
            );
            return;
        }
    }
    // 权限已授予，继续下载
    performDownload();
}

@Override
public void onRequestPermissionsResult(int requestCode, String[] permissions, int[] grantResults) {
    super.onRequestPermissionsResult(requestCode, permissions, grantResults);
    
    if (requestCode == REQUEST_STORAGE_PERMISSION) {
        if (grantResults.length > 0 && grantResults[0] == PackageManager.PERMISSION_GRANTED) {
            performDownload();
        } else {
            showErrorDialog("权限不足", "需要存储权限才能下载文件");
        }
    }
}
```
